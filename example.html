oStat.js

About

The oStat.js library provides access to common statistical and probability tools. It is far from production-ready.

Examples

Getting a random number from a normal distribution with mean 500 and standard deviation 100.
alert(new NormalDistribution(500,100).rand());
Getting 10 random numbers from an exponential distribution with rate 0.5.
alert(new ExponentialDistribution(.5).rand(10));
Getting 100 random numbers from an exponential distribution with rate 0.5. Then finding out what the maximum likelihood estimate for lambda, based on the sample.
var distribution = new ExponentialDistribution(.5);
var rands = distribution.rand(100);
alert(DescriptiveStatistics.maximumLikelihood(rands,distribution).lambda);
Plotting 1000 random variables from a normal distribution.
var distribution = new NormalDistribution(0,1);
var rands = distribution.rand(1000);
DescriptiveStatistics.plotHistogram('#graph_container',rands,{
	title:{text:"Histogram of N(0,1) Random Variates"},
	yAxis:{title:{text:"Count"}}
});
Interactively fit a normal distribution to a set of random data.
var distribution = new NormalDistribution(0,1);
var rands = distribution.rand(1000);
DescriptiveStatistics.plotHistogram('#graph_container',rands, {
	title:{text:"Histogram of N(0,1) Random Variates"},
	yAxis:{title:{text:"Count"}}
});
DescriptiveStatistics.plotInteractivePDF('#graph_container',new NormalDistribution(0,1));
Interactively fit a uniform distribution to a set of random data.
var distribution = new UniformDistribution(0,10);
var rands = distribution.rand(1000);
DescriptiveStatistics.plotHistogram('#graph_container',rands,{
	title:{text:"Histogram of U(0,10) Random Variates"},
	yAxis:{title:{text:"Count"}}
});
DescriptiveStatistics.plotInteractivePDF('#graph_container',new UniformDistribution(0,10));
Interactively fit an exponential distribution to a set of random data.
var distribution = new ExponentialDistribution(.5);
var rands = distribution.rand(1000);
DescriptiveStatistics.plotHistogram('#graph_container',rands,{
	title:{text:"Histogram of Exp(.5) Random Variates"},
	yAxis:{title:{text:"Count"}
});
DescriptiveStatistics.plotInteractivePDF('#graph_container',
	new ExponentialDistribution(DescriptiveStatistics.maximumLikelihood(rands,distribution).lambda));
